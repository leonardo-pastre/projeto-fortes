CREATE TABLE COMBUSTIVEL (
	COMBUSTIVEL_ID INT NOT NULL,
	NOME VARCHAR(50) NOT NULL,
	VALOR NUMERIC (10,2) NOT NULL,
	PRIMARY KEY (COMBUSTIVEL_ID)
);

CREATE TABLE TANQUE (
	TANQUE_ID INT NOT NULL,
	COMBUSTIVEL_ID INT,
	DESCRICAO VARCHAR(50) NOT NULL,
	PRIMARY KEY (TANQUE_ID)
	FOREIGN KEY (COMBUSTIVEL_ID) REFERENCES COMBUSTIVEL (COMBUSTIVEL_ID)
);

CREATE TABLE BOMBA (
	BOMBA_ID INT NOT NULL,
	TANQUE_ID INT NOT NULL,
	COMBUSTIVEL_ID INT NOT NULL,
	DESCRICAO VARCHAR(50) NOT NULL,
	PRIMARY KEY (BOMBA_ID)
	FOREIGN KEY (COMBUSTIVEL_ID) REFERENCES COMBUSTIVEL (COMBUSTIVEL_ID)
);

CREATE TABLE ABASTECIMENTO (
	ABASTECIMENTO_ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	BOMBA_ID INT NOT NULL,
	DATA TEXT NOT NULL,
	QUANTIDADE_COMBUSTIVEL NUMERIC(10,2) NOT NULL,
	VALOR_PAGO NUMERIC(10,2) NOT NULL,
	PERCENTUAL_IMPOSTO NUMERIC(10,2) NULL,
	VALOR_IMPOSTO NUMERIC(10,2) NULL,
	FOREIGN KEY (BOMBA_ID) REFERENCES BOMBA (BOMBA_ID)
);
